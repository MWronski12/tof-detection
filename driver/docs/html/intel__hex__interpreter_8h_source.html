50 /* get the ULBA from a 32-bit address */
51 #define INTEL_HEX_ULBA( adr ) ( (adr) & 0xFFFF0000UL )

53 /* an intel hex record always has at least:
54  :llaaaattcc
55  l= length, a= address, t= type, c= crc
56  so we need at least 11 ascii uint8_tacters for 1 regular record */
57 #define INTEL_HEX_MIN_RECORD_SIZE 11
58 /* the lineLength is at the beginning not the length but the last written
59  address. To make out of this the length we substract instead
60  of the min_record_size the min_last_address. */
61 #define INTEL_HEX_MIN_LAST_ADDRESS ((INTEL_HEX_MIN_RECORD_SIZE) - 1 )
62 
63 /* an intel hex record has a length field in that only 256 = an 8-bit number
64  * can be represented, so we can limit the data buffer to half the size,
65  * as it converts from ascii to binary */
66 #define INTEL_HEX_MAX_RECORD_DATA_SIZE (128)
