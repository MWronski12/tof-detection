92 /* Output buffer is (HEADER_SIZE)+(DATA_SET_SIZE) to account for framing
93  * before sending up the data out of the driver
94  *
95  * *** Driver Output message stream (to-userspace) ***
96  * // Message 1
97  * { Frame ID,
98  * Frame Length
99  * { Frame_Data 0,
100  * Frame_Data 1,
101  * Frame_Data 2,
102  * .
103  * .
104  * .
105  * Frame_Data (size - 1)
106  * }
107  * }
108  * // Message 2
109  * { Frame ID,
110  * Frame Length
111  * { Frame_Data 0,
112  * .
113  * .
114  * .
115  *
116  * User space applications should cast the buffer to the appropriate
117  * structure based off the 'msg_id' see the example below:
118  *
119  * // buffer must be at least 'TMF882X_MAX_MSG_SIZE' large
120  *
121  * uint8_t buf[TMF882X_MAX_MSG_SIZE];
122  * size_t off = 0;
123  * ssize_t ret = 0;
124  * while (1) {
125  * ret = read(fd, buf + off, (sizeof(buf) - off));
126  * off = 0;
127  * do {
128  * struct tmf882x_msg *msg = buf + off;
129  * if (msg->hdr.msg_len > (ret - off)) {
130  * // partial message, shift down and read more
131  * memmove(buf, buf+off, (ret - off));
132  * off = ret - off;
133  * break;
134  * }
135  * off += msg->hdr.msg_len;
136  * switch (msg->hdr.msg_id) {
137  * case ID_ERROR:
138  * struct tmf882x_msg_error *err = &msg->err_msg;
139  * *** Handle msg 'err' ***
140  * break;
141  * case .
142  * .
143  * .
144  * }
145  * } while (off < ret)
146  * }
147  */
