73 #define _IOCTL(mode, nr, dir, isize, osize) \
74  (((dir) << _IOCTL_DIRSHIFT) | \
75  ((mode) << _IOCTL_MODESHIFT) | \
76  ((nr) << _IOCTL_NRSHIFT) | \
77  ((isize) << _IOCTL_ISIZESHIFT) | \
78  ((osize) << _IOCTL_OSIZESHIFT))
79 
80 #define _IOCTL_N(mode, nr) _IOCTL((mode), (nr), _IOCTL_NONE, 0, 0)
81 #define _IOCTL_R(mode, nr, otype) _IOCTL((mode), (nr), _IOCTL_READ, 0, sizeof(otype))
82 #define _IOCTL_W(mode, nr, itype) _IOCTL((mode), (nr), _IOCTL_WRITE, sizeof(itype), 0)
83 #define _IOCTL_RW(mode, nr, itype, otype) _IOCTL((mode), (nr), _IOCTL_READ | _IOCTL_WRITE, sizeof(itype), sizeof(otype))
